рҹ“ң XS йҒёиӮЎи…іжң¬еҺҹе§Ӣзўј вҖ” еӨҡй ӯжӢүеӣһиІ·й»һпјҲеҗ«йҖІеҮәе ҙеғ№пјүпјҲXQ еҸҜзӣҙжҺҘиӨҮиЈҪ вҖ” й»һй–Ӣпјү
// =====================================================================
// еӨҡй ӯжӢүеӣһиІ·й»һ йҒёиӮЎпјҲ3P в’ё еҗ«йҖІеҮәе ҙеғ№В·еҒҡеӨҡзүҲ; еҺҹеҗҚ дёүеұӨжһ¶ж§Ӣж“ҚдҪңзӯ–з•Ҙпјү
//   жңҲпјҡе®ҡеӨҡз©ә  вҶ’ жңҲж”¶зӣӨ >= жңҲMA12 (е№ҙз·ҡ) дё” жңҲMACD DIF > 0
//   йҖұпјҡе®ҡи¶ЁеӢў  вҶ’ 4 й …еҸ– 3пјҡз«ҷдёҠйҖұMA20 / еӨҡй ӯжҺ’еҲ— / еә•еә•й«ҳ / йҖұMACDйҮ‘еҸүдёӯ
//   ж—Ҙпјҡе®ҡиІ·й»һ  вҶ’ жӢүеӣһ MA20 еҸҚеҪҲ + йҮҸеўһпјҲйҰ–йҒёеҒҡеӨҡиЁҠиҷҹпјү
// еҗҢжӯҘијёеҮәпјҡжҲҗдәӨйҮ‘йЎҚ(е„„)гҖҒйҖІе ҙеғ№гҖҒеҒңжҗҚеғ№гҖҒзӣ®жЁҷеғ№гҖҒйўЁе ұжҜ”гҖҒйҖұи©•еҲҶ
// =====================================================================
SetTotalBar(2000);

// === и®Ҡж•ёе®Је‘Ҡ ===
// жңҲз·ҡ MACD
var: monthDif(0), monthMacd(0), monthOsc(0);
// йҖұз·ҡ MACD
var: weekDif(0), weekMacd(0), weekOsc(0);
// еқҮз·ҡ
var: monthMA12(0);
var: weekMA5(0), weekMA10(0), weekMA20(0);
var: ma20(0);
// йҖұжіўж®өдҪҺ
var: wL1(0), wL2(0);
// дёүеұӨжўқд»¶ж——жЁҷ
var: condM(false), condW(false), condD(false);
// йҖұз·ҡеҲҶй …еҫ—еҲҶпјҲ4 й …пјү
var: scoreW(0);
// йҖІеҮәе ҙ
var: entryPrice(0), stopPrice(0), targetPrice(0), rrRatio(0);

// =====================================================================
// иЁҲз®—жүҖжңүжҠҖиЎ“жҢҮжЁҷ
// =====================================================================
MACD(GetField("ж”¶зӣӨеғ№","M"), 12, 26, 9, monthDif, monthMacd, monthOsc);
MACD(GetField("ж”¶зӣӨеғ№","W"), 12, 26, 9, weekDif, weekMacd, weekOsc);

monthMA12 = Average(GetField("ж”¶зӣӨеғ№","M"), 12);
weekMA5   = Average(GetField("ж”¶зӣӨеғ№","W"), 5);
weekMA10  = Average(GetField("ж”¶зӣӨеғ№","W"), 10);
weekMA20  = Average(GetField("ж”¶зӣӨеғ№","W"), 20);
ma20      = Average(Close, 20);

wL1 = Lowest(GetField("жңҖдҪҺеғ№","W"), 8);
wL2 = Lowest(GetField("жңҖдҪҺеғ№","W")[8], 8);

// =====================================================================
// з¬¬дёҖеұӨпјҡжңҲз·ҡе®ҡеӨҡз©ә
// =====================================================================
condM = (Close >= monthMA12) and (monthDif > 0);

// =====================================================================
// з¬¬дәҢеұӨпјҡйҖұз·ҡе®ҡи¶ЁеӢўпјҲ4 й …иҮіе°‘йҒ”жҲҗ 3 й …пјү
// =====================================================================
scoreW = 0;
if Close >= weekMA20 then scoreW = scoreW + 1;                       // з«ҷдёҠйҖұMA20
if (weekMA5 > weekMA10) and (weekMA10 > weekMA20) then scoreW = scoreW + 1;  // еӨҡй ӯжҺ’еҲ—
if wL1 > wL2 then scoreW = scoreW + 1;                                // еә•еә•й«ҳ
if weekDif > weekMacd then scoreW = scoreW + 1;                       // йҖұMACDйҮ‘еҸүдёӯ
condW = scoreW >= 3;

// =====================================================================
// з¬¬дёүеұӨпјҡж—Ҙз·ҡе®ҡиІ·й»һпјҲжӢүеӣһ MA20 еҸҚеҪҲејҸпјү
//   (a) д»Ҡж—Ҙж”¶зӣӨ вүҘ MA20пјҲз«ҷдёҠ MA20пјҢйҒҝе…ҚжҠ“дёӢи·Ңдёӯзҡ„з ҙз·ҡпјү
//   (b) иҝ‘ 3 ж—Ҙе…§жңҖдҪҺжңүи§ёзў° MA20пјҲе®№е·® 1%пјү
//   (c) д»Ҡж—Ҙж”¶зӣӨ > жҳЁж—Ҙж”¶зӣӨпјҲеҸҚеҪҲпјү
//   (d) д»Ҡж—ҘйҮҸ > 5 ж—ҘеқҮйҮҸ Г— 1.2пјҲйҮҸеўһзўәиӘҚпјү
// =====================================================================
condD = (Close >= ma20)
    and (Lowest(Low, 3) <= ma20 * 1.01)
    and (Close > Close[1])
    and (Volume > Average(Volume, 5) * 1.2);

// =====================================================================
// йҖІеҮәе ҙеғ№ж јиЁҲз®—
//   йҖІе ҙеғ№ = з•¶еүҚж”¶зӣӨ
//   еҒңжҗҚеғ№ = max( MA20 Г— 0.99 , йҖІе ҙ Г— 0.93 )  вҶҗ еҸ–ијғиҝ‘иҖ…пјҲжҗҚеӨұијғе°Ҹпјү
//   зӣ®жЁҷеғ№ = йҖІе ҙ Г— 1.30  вҶҗ й җиЁӯзӣ®жЁҷ +30%
//   йўЁе ұжҜ” = (зӣ®жЁҷ - йҖІе ҙ) / (йҖІе ҙ - еҒңжҗҚ)
// =====================================================================
entryPrice  = Close;
stopPrice   = ma20 * 0.99;
if (Close * 0.93) > stopPrice then stopPrice = Close * 0.93;
targetPrice = Close * 1.30;
if (Close - stopPrice) <> 0
    then rrRatio = (targetPrice - Close) / (Close - stopPrice)
    else rrRatio = 0;

// =====================================================================
// е…ұжҢҜи§ёзҷјпјҡдёүеұӨе…Ёз¶ жүҚеӣһеӮі
// =====================================================================
if condM and condW and condD then begin
    ret = 1;
    // order:=1 вҶ’ зөҗжһңй ҒдҫқжҲҗдәӨйҮ‘йЎҚз”ұеӨ§еҲ°е°ҸжҺ’еәҸ
    OutputField1(GetField("жҲҗдәӨйҮ‘йЎҚ(е„„)","D"), "жҲҗдәӨйҮ‘йЎҚ(е„„)", order:=1);
    OutputField2(entryPrice,  "йҖІе ҙеғ№");
    OutputField3(stopPrice,   "еҒңжҗҚеғ№");
    OutputField4(targetPrice, "зӣ®жЁҷеғ№");
    OutputField5(rrRatio,     "йўЁе ұжҜ”");
    OutputField6(scoreW,      "йҖұи©•еҲҶ");
end;
