// =====================================================================
// 三層架構操作策略選股（做多版）
//   月：定多空  → 月收盤 >= 月MA12 (年線) 且 月MACD DIF > 0
//   週：定趨勢  → 4 項取 3：站上週MA20 / 多頭排列 / 底底高 / 週MACD金叉中
//   日：定買點  → 拉回 MA20 反彈 + 量增（首選做多訊號）
// 同步輸出：成交金額(億)、進場價、停損價、目標價、風報比、週評分
// =====================================================================
SetTotalBar(2000);

// === 變數宣告 ===
// 月線 MACD
var: monthDif(0), monthMacd(0), monthOsc(0);
// 週線 MACD
var: weekDif(0), weekMacd(0), weekOsc(0);
// 均線
var: monthMA12(0);
var: weekMA5(0), weekMA10(0), weekMA20(0);
var: ma20(0);
// 週波段低
var: wL1(0), wL2(0);
// 三層條件旗標
var: condM(false), condW(false), condD(false);
// 週線分項得分（4 項）
var: scoreW(0);
// 進出場

// =====================================================================
// 計算所有技術指標
// =====================================================================
MACD(GetField("收盤價","M"), 12, 26, 9, monthDif, monthMacd, monthOsc);
MACD(GetField("收盤價","W"), 12, 26, 9, weekDif, weekMacd, weekOsc);

monthMA12 = Average(GetField("收盤價","M"), 12);
weekMA5   = Average(GetField("收盤價","W"), 5);
weekMA10  = Average(GetField("收盤價","W"), 10);
weekMA20  = Average(GetField("收盤價","W"), 20);

wL1 = Lowest(GetField("最低價","W"), 8);
wL2 = Lowest(GetField("最低價","W")[8], 8);

// 第一層：月線定多空
// =====================================================================
condM = (Close >= monthMA12) and (monthDif > 0);

entryPrice  = Close;

// =====================================================================
// 三層架構選股 — 裸 K 純價量版（做多）
//   月：1 條 → 月收盤 >= 月MA12
//   週：2 條 → 站上週MA20 + 底底高（近 8 週低 > 前 8 週低）
//   日：4 條 → 紅K + 拉回MA20 + 站回MA20 + 量增 1.2x
// 進場/停損(-5%)/目標(+20%)/風報比 一起輸出。依成交金額由大到小排序。
// =====================================================================
SetTotalBar(2000);

// === 變數宣告 ===
var: monthMA12(0);
var: weekMA20(0);
var: wL1(0), wL2(0);
var: ma20(0), avgVol5(0);
var: condM(false), condW(false), condD(false);
var: entryPrice(0), stopPrice(0), targetPrice(0), rrRatio(0);

// === 計算 ===
monthMA12 = Average(GetField("收盤價","M"), 12);
weekMA20  = Average(GetField("收盤價","W"), 20);
wL1 = Lowest(GetField("最低價","W"), 8);
wL2 = Lowest(GetField("最低價","W")[8], 8);
ma20    = Average(Close, 20);
avgVol5 = Average(Volume, 5);

// =====================================================================
// 第一層 月線定多空：1 條
// =====================================================================
condM = Close >= monthMA12;

// =====================================================================
// 第二層 週線定趨勢：2 條全要
//   (a) 站上週MA20
//   (b) 底底高（近 8 週低 > 前 8 週低）
// =====================================================================
condW = (Close >= weekMA20) and (wL1 > wL2);

// =====================================================================
// 第三層 日線定買點：4 條全要
//   (a) 今日紅 K（收 > 開）
//   (b) 近 5 日內有拉回 MA20（容差 1%）
//   (c) 今日收盤站回 MA20
//   (d) 量增（今日量 > 5 日均量 × 1.2）
// =====================================================================
condD = (Close > Open)
    and (Lowest(Low, 5) <= ma20 * 1.01)
    and (Close >= ma20)
    and (Volume > avgVol5 * 1.2);

// =====================================================================
// 進場 / 停損 / 目標
//   進場 = 收盤
//   停損 = max( MA20 × 0.99, 進場 × 0.95 )  ← 取較高（損失較小）
//   目標 = 進場 × 1.20
// =====================================================================
entryPrice  = Close;
stopPrice   = ma20 * 0.99;
if (Close * 0.95) > stopPrice then stopPrice = Close * 0.95;
targetPrice = Close * 1.20;
if (Close - stopPrice) <> 0
    then rrRatio = (targetPrice - Close) / (Close - stopPrice)
    else rrRatio = 0;

// =====================================================================
// 三層全綠才觸發
// =====================================================================
if condM and condW and condD then begin
    ret = 1;
    OutputField1(GetField("成交金額(億)","D"), "成交金額(億)", order:=1);
    OutputField2(entryPrice,  "進場價");
    OutputField3(stopPrice,   "停損價");
    OutputField4(targetPrice, "目標價");
    OutputField5(rrRatio,     "風報比");
end;

三週期框架（CIO 操作法）
月線定多空（大方向：多頭才做多）→ 週線定趨勢（波段慣性：底底高/均線多排）→ 日線定買點（時機：拉回均線的反彈+量能確認）。三層全綠才進場 = 順大勢、擇時機，防「日線漂亮但月線空頭」的接刀。
