рҹ“ң XS йҒёиӮЎи…іжң¬еҺҹе§Ӣзўј вҖ” з«ҷдёҠеқҮз·ҡжӢүеӣһзҙ…KпјҲзҙ”еғ№йҮҸпјүпјҲXQ еҸҜзӣҙжҺҘиӨҮиЈҪ вҖ” й»һй–Ӣпјү
// =====================================================================
// з«ҷдёҠеқҮз·ҡжӢүеӣһзҙ…K йҒёиӮЎпјҲ3P в’№ зҙ”еғ№йҮҸзүҲВ·еҒҡеӨҡ; еҺҹеҗҚ дёүеұӨжһ¶ж§Ӣ иЈёKзүҲпјү
//   жңҲпјҡ1 жўқ вҶ’ жңҲж”¶зӣӨ >= жңҲMA12
//   йҖұпјҡ2 жўқ вҶ’ з«ҷдёҠйҖұMA20 + еә•еә•й«ҳпјҲиҝ‘ 8 йҖұдҪҺ > еүҚ 8 йҖұдҪҺпјү
//   ж—Ҙпјҡ4 жўқ вҶ’ зҙ…K + жӢүеӣһMA20 + з«ҷеӣһMA20 + йҮҸеўһ 1.2x
// йҖІе ҙ/еҒңжҗҚ(-5%)/зӣ®жЁҷ(+20%)/йўЁе ұжҜ” дёҖиө·ијёеҮәгҖӮдҫқжҲҗдәӨйҮ‘йЎҚз”ұеӨ§еҲ°е°ҸжҺ’еәҸгҖӮ
// =====================================================================
SetTotalBar(2000);

// === и®Ҡж•ёе®Је‘Ҡ ===
var: monthMA12(0);
var: weekMA20(0);
var: wL1(0), wL2(0);
var: ma20(0), avgVol5(0);
var: condM(false), condW(false), condD(false);
var: entryPrice(0), stopPrice(0), targetPrice(0), rrRatio(0);

// === иЁҲз®— ===
monthMA12 = Average(GetField("ж”¶зӣӨеғ№","M"), 12);
weekMA20  = Average(GetField("ж”¶зӣӨеғ№","W"), 20);
wL1 = Lowest(GetField("жңҖдҪҺеғ№","W"), 8);
wL2 = Lowest(GetField("жңҖдҪҺеғ№","W")[8], 8);
ma20    = Average(Close, 20);
avgVol5 = Average(Volume, 5);

// =====================================================================
// з¬¬дёҖеұӨ жңҲз·ҡе®ҡеӨҡз©әпјҡ1 жўқ
// =====================================================================
condM = Close >= monthMA12;

// =====================================================================
// з¬¬дәҢеұӨ йҖұз·ҡе®ҡи¶ЁеӢўпјҡ2 жўқе…ЁиҰҒ
//   (a) з«ҷдёҠйҖұMA20
//   (b) еә•еә•й«ҳпјҲиҝ‘ 8 йҖұдҪҺ > еүҚ 8 йҖұдҪҺпјү
// =====================================================================
condW = (Close >= weekMA20) and (wL1 > wL2);

// =====================================================================
// з¬¬дёүеұӨ ж—Ҙз·ҡе®ҡиІ·й»һпјҡ4 жўқе…ЁиҰҒ
//   (a) д»Ҡж—Ҙзҙ… KпјҲж”¶ > й–Ӣпјү
//   (b) иҝ‘ 5 ж—Ҙе…§жңүжӢүеӣһ MA20пјҲе®№е·® 1%пјү
//   (c) д»Ҡж—Ҙж”¶зӣӨз«ҷеӣһ MA20
//   (d) йҮҸеўһпјҲд»Ҡж—ҘйҮҸ > 5 ж—ҘеқҮйҮҸ Г— 1.2пјү
// =====================================================================
condD = (Close > Open)
    and (Lowest(Low, 5) <= ma20 * 1.01)
    and (Close >= ma20)
    and (Volume > avgVol5 * 1.2);

// =====================================================================
// йҖІе ҙ / еҒңжҗҚ / зӣ®жЁҷ
//   йҖІе ҙ = ж”¶зӣӨ
//   еҒңжҗҚ = max( MA20 Г— 0.99, йҖІе ҙ Г— 0.95 )  вҶҗ еҸ–ијғй«ҳпјҲжҗҚеӨұијғе°Ҹпјү
//   зӣ®жЁҷ = йҖІе ҙ Г— 1.20
// =====================================================================
entryPrice  = Close;
stopPrice   = ma20 * 0.99;
if (Close * 0.95) > stopPrice then stopPrice = Close * 0.95;
targetPrice = Close * 1.20;
if (Close - stopPrice) <> 0
    then rrRatio = (targetPrice - Close) / (Close - stopPrice)
    else rrRatio = 0;

// =====================================================================
// дёүеұӨе…Ёз¶ жүҚи§ёзҷј
// =====================================================================
if condM and condW and condD then begin
    ret = 1;
    OutputField1(GetField("жҲҗдәӨйҮ‘йЎҚ(е„„)","D"), "жҲҗдәӨйҮ‘йЎҚ(е„„)", order:=1);
    OutputField2(entryPrice,  "йҖІе ҙеғ№");
    OutputField3(stopPrice,   "еҒңжҗҚеғ№");
    OutputField4(targetPrice, "зӣ®жЁҷеғ№");
    OutputField5(rrRatio,     "йўЁе ұжҜ”");
end;
